-- Q1 Active Enrollments with Student and Course Details. INNER JOIN with WHERE and ORDER BY
SELECT 
    s.student_id,
    CONCAT(s.first_name, ' ', s.last_name) AS student_name,
    c.course_title,
    e.enrollment_date,
    e.status
FROM Enrollments e
INNER JOIN Students s ON e.student_id = s.student_id
INNER JOIN Courses c ON e.course_id = c.course_id
WHERE e.status = 'Active'  -- Filter for active enrollments only
ORDER BY e.enrollment_date DESC;

-- Q2 Total Revenue by Course Category. INNER JOIN with Aggregate Functions and GROUP BY
SELECT 
    c.category,
    COUNT(e.enrollment_id) AS total_enrollments,
    SUM(c.price) AS total_revenue,
    AVG(c.price) AS avg_course_price
FROM Courses c
INNER JOIN Enrollments e ON c.course_id = e.course_id
WHERE c.price > 0  -- Exclude free courses
GROUP BY c.category
ORDER BY total_revenue DESC;

-- Q 4 Students with Below Average Performance. LEFT JOIN with Aggregate and HAVING Clause.
SELECT 
    s.student_id,
    CONCAT(s.first_name, ' ', s.last_name) AS student_name,
    c.course_title,
    AVG(sub.score) AS student_avg_score,
    (SELECT AVG(score) FROM Submissions) AS overall_avg_score
FROM Students s
LEFT JOIN Submissions sub ON s.student_id = sub.student_id
LEFT JOIN Assignments a ON sub.assignment_id = a.assignment_id
LEFT JOIN Courses c ON a.course_id = c.course_id
GROUP BY s.student_id, c.course_id
HAVING student_avg_score < (SELECT AVG(score) FROM Submissions)
   AND student_avg_score IS NOT NULL;

-- Q8   Top Performers in SQL Course (Above Average).  Subquery in WHERE Clause.
SELECT 
    s.student_id,
    CONCAT(s.first_name, ' ', s.last_name) AS student_name,
    sub.score,
    a.title AS assignment_title
FROM Students s
JOIN Submissions sub ON s.student_id = sub.student_id
JOIN Assignments a ON sub.assignment_id = a.assignment_id
JOIN Courses c ON a.course_id = c.course_id
WHERE c.course_title = 'Introduction to SQL'
  AND sub.score > (
      SELECT AVG(score) 
      FROM Submissions sub2
      JOIN Assignments a2 ON sub2.assignment_id = a2.assignment_id
      JOIN Courses c2 ON a2.course_id = c2.course_id
      WHERE c2.course_title = 'Introduction to SQL'
  )
ORDER BY sub.score DESC
LIMIT 10;  -- Show top 10 performers

-- Q9 Student Performance vs Course Average. ubquery in SELECT Clause with Aggregates.
SELECT 
    s.student_id,
    CONCAT(s.first_name, ' ', s.last_name) AS student_name,
    c.course_title,
    AVG(sub.score) AS student_avg_score,
    (SELECT AVG(sub2.score) 
     FROM Submissions sub2
     JOIN Assignments a2 ON sub2.assignment_id = a2.assignment_id
     WHERE a2.course_id = c.course_id) AS course_avg_score,
    AVG(sub.score) - (SELECT AVG(sub2.score) 
                      FROM Submissions sub2
                      JOIN Assignments a2 ON sub2.assignment_id = a2.assignment_id
                      WHERE a2.course_id = c.course_id) AS score_difference
FROM Students s
JOIN Submissions sub ON s.student_id = sub.student_id
JOIN Assignments a ON sub.assignment_id = a.assignment_id
JOIN Courses c ON a.course_id = c.course_id
GROUP BY s.student_id, c.course_id
HAVING score_difference IS NOT NULL
ORDER BY score_difference DESC;

-- Q10 Courses with Low Submission Rates Despite High Enrollment with Multiple JOINs and HAVING.
SELECT 
    c.course_id,
    c.course_title,
    COUNT(DISTINCT e.student_id) AS enrolled_students,
    COUNT(DISTINCT sub.submission_id) AS total_submissions,
    COUNT(DISTINCT a.assignment_id) AS total_assignments,
    ROUND(COUNT(DISTINCT sub.submission_id) * 100.0 / 
          (COUNT(DISTINCT e.student_id) * COUNT(DISTINCT a.assignment_id)), 2) AS submission_rate_percent
FROM Courses c
LEFT JOIN Enrollments e ON c.course_id = e.course_id
LEFT JOIN Assignments a ON c.course_id = a.course_id
LEFT JOIN Submissions sub ON a.assignment_id = sub.assignment_id
WHERE e.status = 'Active'
GROUP BY c.course_id
HAVING enrolled_students > 2  -- Courses with more than 2 enrolled students
   AND submission_rate_percent < 50.0  -- Less than 50% submission rate
   AND total_assignments > 0  -- Has at least one assignment
ORDER BY submission_rate_percent ASC;
